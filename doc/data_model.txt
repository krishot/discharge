
# Base generator for Doc Say's App
#   
rake db:drop 
rake db:create       
rm db/migrate/*

# Patients have a proxy model called patient

rails g scaffold Patient first_name:string last_name:string email:string phone:string description:text state:string  engaged_at:datetime last_contact:datetime carepass_id:string mhealth_id:string physician:references conditions:references actions:references alerts:references

rails g scaffold Physician first_name:string last_name:string email:string phone:string specialty:text state:string   patients:references

rails g scaffold Condition name:string description:text url:text state:string type:string category:string icd9:string action_program:references offers:references     

rails g scaffold ActionProgram name:string category:string description:text state:string type:string therapy_id:string condition:references

rails g scaffold ActionTask name:string description:text url:string state:string type:string category:string action_program:references     

rails g scaffold PatientTask name:string state:string action_task:references patient:references

#
#  Need to run migration and bootstrap 
rake db:migrate
rails g bootstrap:themed Patients
rails g bootstrap:themed Physicians
rails g bootstrap:themed Conditions
rails g bootstrap:themed ActionPrograms
rails g bootstrap:themed ActionTasks  
rails g bootstrap:themed PatientTasks
